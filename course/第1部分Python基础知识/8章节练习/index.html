'''每个数字可以写成多个质数的乘积，给定一个数字，请你分解为多个质数'''

def fun(num, list=None):
    if list is None:
        list = []
    for i in range(2, num):
        while num % i == 0:
            list.append(i)
            num = int(num / i)
            if num > 1:
                fun(num)
    return list
x = 9*5
print(fun(x))

'''
编写一段代码，模拟我们数羊到入睡的过程：


每数五只羊，就提问一次：睡着了吗？
如果没有睡着，继续循环，并打印“继续数羊”。
如果睡着了，则停止循环，并打印“终于睡着了”。
'''


i = 0
while True:
    i += 1
    left_endpoint = 1 + 5 * ( i - 1 )
    right_endpoint = 1 + 5 * i
    for i in range(left_endpoint, right_endpoint):
        print(str(i)+'只羊')
    answer = input('睡着了吗？回答是或否：')
    if answer == '是':
        break
    print('继续数羊')
print('终于睡着了')
